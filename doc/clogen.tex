\begin{algorithm}[t]
\caption{get\_first\_parent}
\label{algo_gfp}
\begin{algorithmic}[1]
  \REQUIRE pattern $P, P \neq \bot $\AND$ P \neq \clo(\bot) $
  \STATE $Z := P$
  \WHILE {$Z \neq \emptyset$}
  \STATE $e = $get\_tail\_element($Z$)
  \STATE $X := Z \setminus \{e\}$
  \IF{ $\clo(X) \neq P $}
  \RETURN $(\clo(X), e$)
  \ENDIF
  \STATE $Z := X$
  \ENDWHILE
  \STATE
  \COMMENT Never Reached
\end{algorithmic}
\end{algorithm}

\begin{algorithm}[t]
\caption{get\_tail\_element}
\label{algo_gte}
\begin{algorithmic}[1]
  \REQUIRE pattern $P \in \F$
  \STATE $max_e := nil$
  \FORALL  {$e \in P$}
  \IF {$\M(P \setminus \{e\}) = 1$}
  \IF {$max_e = nil$ \OR $e > max_e$}
  \STATE $max_e := e$
  \ENDIF 
  \ENDIF  
  \ENDFOR
  \RETURN $max_e$
\end{algorithmic}
\end{algorithm}
